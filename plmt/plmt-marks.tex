\setstcolor{del}
\setul{0.2pt}{1pt}
\setulcolor{hlh}

\newcommand\edmark[1]{\small\color{meta}\bfseries\sffamily #1}

\newcommand\add[2][]{%
\ifstrempty{#1}{%
\textcolor{add}{#2}%
}{%
\textcolor{add}{#2 \edmark{#1}}%
}%
}

\newcommand\remove[2][]{%
\ifstrempty{#1}{%
\textcolor{meta}{\st{#2}}%
}{%
\textcolor{meta}{\st{#2} \edmark{#1}}%
}%
}

\newcommand\note[2][]{%
\ifstrempty{#1}{%
\textcolor{accent}{[\small\sffamily #2]}%
}{%
\textcolor{accent}{[\small\sffamily #2 -- \edmark{#1}]}%
}%
}

\newcommand{\highlight}[1]{\ul{#1}}
